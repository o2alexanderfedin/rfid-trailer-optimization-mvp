class AsyncQueue<T> {
  constructor(maxSize = 1)               // maxSize >= 1; power-of-2 circular buffer
  async enqueue(item: T): Promise<void>  // BLOCKS producer when count >= maxSize (backpressure); throws if closed
  async dequeue(): Promise<T | undefined> // BLOCKS consumer when empty; returns undefined after close+drain
  close(): void                          // wakes blocked waiters; consumers may still drain remaining items
  get size(): number; get isFull(): boolean; get isEmpty(): boolean;
}
