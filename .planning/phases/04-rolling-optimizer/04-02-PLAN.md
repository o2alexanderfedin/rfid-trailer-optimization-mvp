<interfaces>
Define in src/graph/types.ts (the Wave-2 consumer contract — keep stable):

```
export type EdgeKind = "trip" | "wait" | "crossDock" | "load" | "unload" | "hold";
export interface FlowNode { readonly id: string; readonly hubId: string; readonly timeMin: number; } // timeMin = minutes from horizon start
export interface FlowEdge {
  readonly id: string;
  readonly from: string;   // FlowNode.id (tail)
  readonly to: string;     // FlowNode.id (head)
  readonly kind: EdgeKind;
  readonly capacity: number;   // integer
  readonly cost: number;       // integer (scaled) per unit flow
}
export interface TimeExpandedGraph {
  readonly nodes: readonly FlowNode[];   // deterministic order
  readonly edges: readonly FlowEdge[];   // deterministic order
  readonly nodeIndex: ReadonlyMap<string, FlowNode>;
}
export interface OptimizerScope { readonly hubIds: readonly string[]; readonly trailerIds: readonly string[]; readonly horizonStartMin: number; readonly horizonEndMin: number; readonly timeStepMin: number; }
```
Node id convention: `${hubId}@${timeMin}`. Use a fixed `timeStepMin` (coarse 15-min nodes per P9) so the graph stays small.
