LruMap implementation (from RESEARCH Pattern 4 + PATTERNS pattern for lru-map.ts):
```typescript
export class LruMap<K, V> {
  private readonly cap: number;
  private readonly map = new Map<K, V>(); // insertion order = LRU order (ES6 spec-guaranteed)
  constructor(cap: number) { this.cap = cap; }
  get(k: K): V | undefined {
    const v = this.map.get(k);
    if (v !== undefined) {
      this.map.delete(k);   // remove from current position
      this.map.set(k, v);   // re-insert at end (MRU)
    }
    return v;
  }
  set(k: K, v: V): void {
    if (this.map.has(k)) this.map.delete(k);   // remove if exists (avoid duplicate)
    this.map.set(k, v);                          // insert at end (MRU)
    if (this.map.size > this.cap) {
      // Evict LRU entry: first key in insertion order (oldest)
      this.map.delete(this.map.keys().next().value as K);
    }
  }
  get size(): number { return this.map.size; }
}
```
